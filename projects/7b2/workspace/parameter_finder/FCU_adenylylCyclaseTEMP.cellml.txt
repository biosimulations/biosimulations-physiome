    case (time < stimSt) and (time > stimSt-tR):   
        stimHolding+m*(time-stimSt+tR);
    case (time >= stimSt) and (time < stimSt+stimDur): 
        stimMag+stimHolding;   
    case (time < stimSt+tR+stimDur) and (time >= stimSt+stimDur):  
        stimHolding+-m*(time-stimSt-tR-stimDur);   
    otherwise:             
        stimHolding;       
endsel;
